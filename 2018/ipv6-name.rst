    # -*- coding: utf-8 -*-

    import re
    import sys
    from io import open

    context = sys.argv[1]
    fha = re.compile(r"^[a-fA-F0-9]{4}\n")
    with open(context, encoding='utf8', errors='ignore') as lines:
    for line in lines:
